        let timerActive = false;

        let nextBreakTime = null;

        let countdownInterval = null;

        let notificationTimeout = null;

        let deferredPrompt = null;


        // PWA Installation

        window.addEventListener('beforeinstallprompt', (e) => {

            e.preventDefault();

            deferredPrompt = e;

            document.getElementById('installPrompt').style.display = 'block';

        });


        function installApp() {

            if (deferredPrompt) {

                deferredPrompt.prompt();

                deferredPrompt.userChoice.then((choiceResult) => {

                    if (choiceResult.outcome === 'accepted') {

                        document.getElementById('installPrompt').style.display = 'none';

                    }

                    deferredPrompt = null;

                });

            }

        }


        // Request notification permission

        if ('Notification' in window) {

            Notification.requestPermission();

        }


        function updateCurrentTime() {

            const now = new Date();

            const timeString = now.toLocaleTimeString('en-US', { 

                hour12: false, 

                hour: '2-digit', 

                minute: '2-digit', 

                second: '2-digit' 

            });

            document.getElementById('currentTime').textContent = timeString;

        }


        function isWithinWorkHours() {

            const now = new Date();

            const startTime = document.getElementById('startTime').value;

            const endTime = document.getElementById('endTime').value;

            

            const [startHour, startMin] = startTime.split(':').map(Number);

            const [endHour, endMin] = endTime.split(':').map(Number);

            

            const currentHour = now.getHours();

            const currentMin = now.getMinutes();

            

            const currentMinutes = currentHour * 60 + currentMin;

            const startMinutes = startHour * 60 + startMin;

            const endMinutes = endHour * 60 + endMin;

            

            return currentMinutes >= startMinutes && currentMinutes < endMinutes;

        }


        function calculateNextBreakTime() {

            const now = new Date();

            const nextHour = new Date(now);

            nextHour.setMinutes(0);

            nextHour.setSeconds(0);

            nextHour.setMilliseconds(0);

            nextHour.setHours(nextHour.getHours() + 1);

            

            // If next hour is beyond work hours, set to start of next work day

            const endTime = document.getElementById('endTime').value;

            const [endHour] = endTime.split(':').map(Number);

            

            if (nextHour.getHours() > endHour) {

                const startTime = document.getElementById('startTime').value;

                const [startHour, startMin] = startTime.split(':').map(Number);

                

                nextHour.setDate(nextHour.getDate() + 1);

                nextHour.setHours(startHour);

                nextHour.setMinutes(startMin);

            }

            

            return nextHour;

        }


        function updateCountdown() {

            if (!timerActive || !nextBreakTime) return;

            

            const now = new Date();

            const timeUntilBreak = nextBreakTime - now;

            

            if (timeUntilBreak <= 0) {

                if (isWithinWorkHours()) {

                    showBreakNotification();

                }

                nextBreakTime = calculateNextBreakTime();

                return;

            }

            

            const hours = Math.floor(timeUntilBreak / (1000 * 60 * 60));

            const minutes = Math.floor((timeUntilBreak % (1000 * 60 * 60)) / (1000 * 60));

            const seconds = Math.floor((timeUntilBreak % (1000 * 60)) / 1000);

            

            const countdownText = hours > 0 

                ? `${hours}h ${minutes}m ${seconds}s`

                : `${minutes}m ${seconds}s`;

                

            document.getElementById('countdown').textContent = countdownText;

        }

            // Visual notification

            document.getElementById('breakNotification').style.display = 'block';

            

            // Browser notification

            if ('Notification' in window && Notification.permission === 'granted') {

                    icon: 'data:image/svg+xml;base64,PHN2ZyB4bWxucz0iaHR0cDovL3d3dy53My5vcmcvMjAwMC9zdmciIHZpZXdCb3g9IjAgMCA5NiA5NiIgZmlsbD0iIzNmNjFmYSI+PGNpcmNsZSBjeD0iNDgiIGN5PSI0OCIgcj0iMTUiIGZpbGw9IiNmZmZmZmYiLz48dGV4dCB4PSI0OCIgeT0iNTYiIHRleHQtYW5jaG9yPSJtaWRkbGUiIGZpbGw9IiNjNDIxMjEiIGZvbnQtc2l6ZT0iMjAiPlJFU1Q8L3RleHQ+PC9zdmc+',

                    requireInteraction: true

                });

            }

            

            // Auto-dismiss after 30 seconds if not manually dismissed

            notificationTimeout = setTimeout(() => {

                dismissBreak();

            }, 30000);

        }


        function dismissBreak() {

            document.getElementById('breakNotification').style.display = 'none';

            if (notificationTimeout) {

                clearTimeout(notificationTimeout);

                notificationTimeout = null;

            }

        }


        function startTimer() {

            timerActive = true;

            nextBreakTime = calculateNextBreakTime();

            

            document.getElementById('status').textContent = 'Timer Active - Break reminders every hour';

            document.getElementById('nextBreak').style.display = 'block';

            document.getElementById('startBtn').style.display = 'none';

            document.getElementById('stopBtn').style.display = 'block';

            

            // Update countdown every second

            countdownInterval = setInterval(() => {

                updateCountdown();

            }, 1000);

            

            // Save state

            localStorage.setItem('breakTimerActive', 'true');

            localStorage.setItem('nextBreakTime', nextBreakTime.getTime());

        }


        function stopTimer() {

            timerActive = false;

            nextBreakTime = null;

            

            document.getElementById('status').textContent = 'Timer Not Active';

            document.getElementById('nextBreak').style.display = 'none';

            document.getElementById('startBtn').style.display = 'block';

            document.getElementById('stopBtn').style.display = 'none';

            

            if (countdownInterval) {

                clearInterval(countdownInterval);

                countdownInterval = null;

            }

            

            dismissBreak();

            

            // Clear saved state

            localStorage.removeItem('breakTimerActive');

            localStorage.removeItem('nextBreakTime');

        }


        // Initialize app

        function initApp() {

            updateCurrentTime();

            setInterval(updateCurrentTime, 1000);

            

            // Load saved state

            const savedActive = localStorage.getItem('breakTimerActive');

            const savedNextBreakTime = localStorage.getItem('nextBreakTime');

            

            if (savedActive === 'true' && savedNextBreakTime) {

                const savedTime = new Date(parseInt(savedNextBreakTime));

                if (savedTime > new Date()) {

                    timerActive = true;

                    nextBreakTime = savedTime;

                    document.getElementById('status').textContent = 'Timer Active - Break reminders every hour';

                    document.getElementById('nextBreak').style.display = 'block';

                    document.getElementById('startBtn').style.display = 'none';

                    document.getElementById('stopBtn').style.display = 'block';

                    

                    countdownInterval = setInterval(() => {

                        updateCountdown();

                    }, 1000);

                } else {

                    // Saved time has passed, clear it

                    localStorage.removeItem('breakTimerActive');

                    localStorage.removeItem('nextBreakTime');

                }

            }

        }

        if ('serviceWorker' in navigator) {

        }


        initApp();

    </script>
